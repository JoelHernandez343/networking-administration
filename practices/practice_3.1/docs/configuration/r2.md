Configuration for R2 (just copy and paste):
```
# ssh and user configuration

config t

ip domain-name adminredes.escom.ipn.mx

ip ssh rsa keypair-name sshkey
crypto key generate rsa usage-keys label sshkey modulus 1024

ip ssh v 2
ip ssh time-out 30
ip ssh authentication-retries 3
line vty 0 15
password cisco
login local
transport input ssh telnet
exit

username admin privilege 15 password admin

# Interfaces

int f 0/0
ip add 192.168.0.97 255.255.255.224
no shu

int f 0/1
ip add 192.168.0.162 255.255.255.252
no shu

int f 1/0
ip add 192.168.0.170 255.255.255.252
no shu

int loop 0
ip add 192.168.10.2 255.255.255.0
no shu

exit

# OSPF

router ospf 1
net 192.168.0.96 0.0.0.31 area 0
net 192.168.0.160 0.0.0.3 area 0
net 192.168.0.168 0.0.0.3 area 0

exit

# SNMP

snmp-server com ro_4CM1 ro
snmp-server com rw_4CM1 rw

snmp-server contact joel.programador@gmail.com
snmp-server location sectionB

exit

```
